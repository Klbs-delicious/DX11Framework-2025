ProjectRoot/
├── Include/                  # C++のヘッダー（宣言・インターフェース）
│   ├── Core/                # システム系（WindowSystem, D3D11Systemなど）
│   ├── Graphics/            # Mesh, Texture, Rendererなど
│   ├── Shaders/             # Shader定数構造体など（C++とHLSLの橋渡し）
│   └── Utils/               # ログ、非コピー化、ヘルパーなど

├── Source/                   # C++の実装（ロジック・処理）
│   ├── Core/
│   ├── Graphics/
│   ├── Shaders/             # ShaderManager, Compilerなど
│   ├── Scenes/              # シーン制御・遷移
│   ├── Entities/            # ゲームオブジェクト
│   └── Utils/

├── Shaders/                  # GPU用コード（HLSLファイル）
│   ├── Framework/           # 再利用可能な描画ロジック
│   └── GameSpecific/        # ゲーム固有の演出・UI表現

├── Assets/                   # 実行時リソース（素材・バイナリ）
│   ├── Shaders/             # コンパイル済み .cso ファイル
│   ├── Textures/            # .png, .dds など画像素材
│   ├── Models/              # .fbx, .obj など立体モデル
│   ├── Sounds/              # .wav, .mp3 など音声素材
│   └── Fonts/               # .ttf などフォント定義

└── External/                 # 外部ライブラリ・SDKなど
